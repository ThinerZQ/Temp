90度旋转数组

             public static void reverse(int [][] matrix){
                         int length = matrix. length;
                         int l = length;
                         for(int i = 0; i < length/ 2; i ++){
                                     for(int j = 0; j < length- 1-i -i; j++){                                   
                                                 int temp = matrix[ i][j +i]; //左上
                                                 matrix[i ][j+ i] = matrix [j+ i][length -1- i];//左上等于右上
                                                 matrix[j +i][ length-1 -i] = matrix[length -1- i][length -1- i-j ];//右上等于右下
                                                 matrix[length -1- i][length -1- i-j ] = matrix[ length-1 -i- j][i ];//右下等于左下
                                                 matrix[length -1- i-j ][i] = temp;                 
                                    }
                        }
            }
