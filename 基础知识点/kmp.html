             public static void get_next(int [] next, char[] pat , int l) {
                         int i = 0;
                         int j = -1;
                         next[0 ] = -1;
                         while (i < l - 1) {
                                     if (j == -1 || pat[i ] == pat[ j]) {
                                                 i++;
                                                 j++;
                                                 if (pat [i] != pat[j ])
                                                             next[i ] = j;
                                                 else
                                                             next[i ] = next[ j];
                                    } else {
                                                 j = next [j];
                                    }
                        }
            }

             public static int kmp(char [] pat, char[] src , int[] next) {
                         int i = 0;
                         int j = 0;
                         while (i < src. length && j < pat. length) {
                                     if (j == -1 || pat[j ] == src[ i]) {
                                                 i++;
                                                 j++;
                                    } else {
                                                 j = next [j];
                                    }
                        }
                         if (j >= pat. length) {
                                     return i - pat. length;
                        } else {
                                     return -1 ;
                        }
            }
